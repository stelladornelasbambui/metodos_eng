        <select id="titulacao" required>
            <option value="graduacao">Graduação</option>

            <option value="mestrado">Mestrado</option>

    <div id="resultado"></div>

    <script>
        document.getElementById("calcularSalario").onclick = function() {
            // Obtendo os valores do formulário
            var nome = document.getElementById("nome").value;
            var endereco = document.getElementById("endereco").value;
            var dataNascimento = document.getElementById("dataNascimento").value;
            var salarioBase = parseFloat(document.getElementById("salarioBase").value);
            var anoInicio = parseInt(document.getElementById("anoInicio").value);
            var titulacao = document.getElementById("titulacao").value;

            // Calculando os anos trabalhados
            var anoAtual = new Date().getFullYear();
            var anosTrabalho = anoAtual - anoInicio;

            // Calculando o aumento por tempo de trabalho (2% por ano)
            var salarioComAumento = salarioBase * (1 + (anosTrabalho * 0.02));

            // Calculando a gratificação com base na titulação
            var gratificacao = 0;
            switch (titulacao) {
                case "pos":
                    gratificacao = 0.05; // 5% para pós-graduação
                    break;
                case "mestrado":
                    gratificacao = 0.10; // 10% para mestrado
                    break;
                case "doutorado":
                    gratificacao = 0.15; // 15% para doutorado
                    break;
                default:
                    gratificacao = 0; // Nenhuma gratificação para graduação
                    break;
            }

            // Aplica a gratificação ao salário com aumento
            var salarioComGratificacao = salarioComAumento * (1 + gratificacao);

            // Calcula o desconto do INSS (4%)
            var descontoINSS = salarioComGratificacao * 0.04;
            var salarioFinal = salarioComGratificacao - descontoINSS;

            // Exibe o resultado na página
            document.getElementById("resultado").innerHTML =
                "<p><strong>Nome:</strong> " + nome + "<br>" +
                "<strong>Endereço:</strong> " + endereco + "<br>" +
                "<strong>Data de Nascimento:</strong> " + dataNascimento + "<br>" +
                "<strong>Salário Base:</strong> R$ " + salarioBase.toFixed(2) + "<br>" +
                "<strong>Anos de Trabalho:</strong> " + anosTrabalho + "<br>" +
                "<strong>Salário com Aumento:</strong> R$ " + salarioComAumento.toFixed(2) + "<br>" +
                "<strong>Gratificação de Titulação:</strong> " + (gratificacao * 100) + "%<br>" +
                "<strong>Salário com Gratificação:</strong> R$ " + salarioComGratificacao.toFixed(2) + "<br>" +
                "<strong>Desconto INSS (4%):</strong> R$ " + descontoINSS.toFixed(2) + "<br>" +
                "<strong>Salário Final:</strong> R$ " + salarioFinal.toFixed(2) + "</p>";
        };
    </script>
